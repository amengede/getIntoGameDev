#version 330 core

const vec3[3] positions = vec3[](
    vec3(-0.75, -0.75, 0.0),
    vec3(0.75, -0.75, 0.0),
    vec3(0.0, 0.75, 0.0)
);

const vec3[3] colors = vec3[](
    vec3(1.0, 0.0, 0.0),
    vec3(0.0, 1.0, 0.0),
    vec3(0.0, 0.0, 1.0)
);

out vec3 fragmentColor;

void main()
{
    gl_Position = vec4(positions[gl_VertexID], 1.0);
    fragmentColor = colors[gl_VertexID];
}
